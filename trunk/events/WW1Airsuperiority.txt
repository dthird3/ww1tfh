
#AIRSUP1

 country_event = {
	title = "AIRSUP1"
	desc = "desc_AIRSUP1"
	id = 777777
	 
	picture = "AIRSUP1"
	 
	mean_time_to_happen = {
		 days = 1
		 }
	trigger = {
		 THIS = {
			 faction = allies
			 }
		 OR = {
			 GER = {
				 total_we_bomb = 5
				 war_with = THIS
				 }
			 AUH = {
				 total_we_bomb = 5
				 war_with = THIS
				 }
			 TUR = {
				 total_we_bomb = 5
				 war_with = THIS
				 }
			 BUL = {
				 total_we_bomb = 5
				 war_with = THIS
				 }
			 }
		 NOT = {
			 THIS = {
				 any_owned_province = {
					 has_province_modifier = scouted
					 }
				 }
			 }
		 NOT = {
			has_country_modifier = scouted
		 }
		 }
	option = {
		 name = "opta_AIRSUP1"
		 ai_chance = {
			 factor = 100
			 }
		 THIS = {
			 any_owned = {
				 limit = {
					 NOT = {
						 has_province_modifier = scouted
						 }
					 any_neighbor_province = {
						 OR = {
							 controlled_by = GER
							 controlled_by = AUH
							 controlled_by = TUR
							 controlled_by = BUL
							 }
						 }
					 }
				 add_province_modifier = {
					 name = scouted
					 duration = 30
					 }
				 }
				add_country_modifier = {
					 name = scouted
					 duration = 30
				}
			 }
		 }
	}

country_event = {

		id = 8001
		

		trigger = {
			faction = axis
			war_with = RUS
			not = { has_country_flag = russia_wargoal }
			not = { has_wargoal = RUS }
		}
		
		mean_time_to_happen = {
			days = 1
		}

		fire_only_once = yes

		title = "WargoalE1"
		desc = "desc_WargoalE1"
		picture = "newspaper"

		option = {
			name = "opta_WargoalE1"
			ai_chance = { factor = 100 }
			set_country_flag = russia_wargoal
			add_wargoal = {
				target = RUS
				war_goal = { casus_belli = axis_russia }
			}
		}
		option = {
			name = "optb_WargoalE1"
			ai_chance = { factor = 0 }
			set_country_flag = russia_wargoal
		}
}
